Quick Start 

use armature::prelude::*;

#[injectable]
struct UserService;

impl UserService {
    fn new() -> Self { Self }
    fn get_users(&self) -> Vec<String> {
        vec!["Alice".to_string(), "Bob".to_string()]
    }
}

#[controller("/api")]
struct UserController {
    service: UserService,
}

impl UserController {
    fn new(service: UserService) -> Self {
        Self { service }
    }

    #[get("/users")]
    fn list(&self, _req: HttpRequest) -> Result<HttpResponse, Error> {
        let users = self.service.get_users();
        Ok(HttpResponse::ok().with_json(&users)?)
    }
}

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let mut app = Application::new();
    app.listen("127.0.0.1:3000").await?;
    Ok(())
}

Installation

Add to your Cargo.toml 

[dependencies]
armature = "0.1"
tokio = { version = "1.35", features = ["full"] }

Run your app 

cargo run

Your server is now running at http://127.0.0.1:3000
