Clean, Expressive Syntax

Familiar patterns for developers coming from Angular or NestJS

main.rs 
use armature::prelude::*;

#[controller("/api")]
struct ApiController;

#[get("/hello/:name")]
async fn hello(&self, name: Path<String>) -> Json<Message> {
    Json(Message { text: format!("Hello, {}!", name.0) })
}

#[module]
struct AppModule {
    #[controller]
    api: ApiController,
}

#[tokio::main]
async fn main() -> Result<()> {
    Application::create::<AppModule>()
        .await
        .listen(3000)
        .await
}
